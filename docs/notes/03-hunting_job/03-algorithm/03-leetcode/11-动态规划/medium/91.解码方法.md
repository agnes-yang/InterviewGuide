## 91. 解码方法

[力扣原题链接（点我直达）](https://leetcode-cn.com/problems/decode-ways/)

一条包含字母 `A-Z` 的消息通过以下方式进行了编码：

```
'A' -> 1
'B' -> 2
...
'Z' -> 26
```

给定一个只包含数字的**非空**字符串，请计算解码方法的总数。

**示例 1:**

```
输入: "12"
输出: 2
解释: 它可以解码为 "AB"（1 2）或者 "L"（12）。
```

**示例 2:**

```
输入: "226"
输出: 3
解释: 它可以解码为 "BZ" (2 26), "VF" (22 6), 或者 "BBF" (2 2 6) 。
```

```c++
int numDecodings(string s) {
	int len = s.size();
	if (s[0] == '0') return 0; //首字母为0，肯定是不能编码的
	if (len == 1) return 1;
	vector<int>  dp(len, 0);
	dp[0] = 1;
	if ((s[0] == '1' || s[0] == '2') && s[1] > '0' && s[1]<'7') dp[1] = 2;
	else
		dp[1] = 1;
	for (int i = 2; i < len; ++i) {
		if (s[i] == '0' && s[i - 1] == '0') return 0;
		if (s[i] == '0') {
			if (s[i - 1] == '1' || s[i - 1] == '2')  	dp[i] = dp[i - 2];
			else
				return 0;
		}

		else if (s[i - 1] == '1') dp[i] = dp[i - 1] + dp[i - 2];

		else
			if (s[i - 1] == '2' && s[i] >= '1' && s[i] <= '6')   dp[i] = dp[i - 1] + dp[i - 2];
		else
				dp[i] = dp[i - 1];

	}

	return dp[len - 1];
}
```

```c++
int numDecodings(string s) {
	int len = s.size();
	if (len == 0 || (len == 1 && s[0] == '0')) return 0;
	if (len == 1) return 1;
	vector<int> dp(len + 1, 0);
	dp[0] = 1;
	for (int i = 0; i < len; ++i) {
		dp[i + 1] = s[i] == '0' ? 0 : dp[i];
		if (i > 0 && (s[i - 1] == '1' || (s[i - 1] == '2' && s[i] <= '6'))) {
			dp[i + 1] += dp[i - 1];
		}
	}
	return dp[len];
}
```
